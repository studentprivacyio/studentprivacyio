export const generateEmail = ({ studentName, parentName, softwareList }) => {
    const softwareListText = softwareList.map(software => 
        software.company ? `${software.name} by ${software.company}` : software.name
    ).join(', ');

    const isMultiple = softwareList.length > 1;
    const subject = isMultiple ? 'PPRA Opt-Out Request - Multiple Software' : `PPRA Opt-Out Request - ${softwareList[0].name}`;

    return `Subject: ${subject}
Dear School Administrator,

I, ${parentName}, as the parent/legal guardian of ${studentName}, am writing to exercise my rights under the Protection of Pupil Rights Amendment (PPRA) regarding the use of ${isMultiple ? 'the following software tools' : 'this software'}: ${softwareListText}.

I hereby opt out of and deny permission for my child to participate in:

1. Any evaluation or survey concerning protected information categories through ${isMultiple ? 'these software tools' : 'this software'}
2. Any marketing surveys or evaluations conducted via ${isMultiple ? 'these platforms' : 'this platform'}
3. Any non-emergency, invasive data collection or monitoring through ${isMultiple ? 'these systems' : 'this system'}

This opt-out should be recorded and honored for the current school year and all future academic years unless explicitly revoked by me in writing. Please ensure that ${isMultiple ? 'all relevant software companies are' : 'the software company is'} notified of this opt-out request.

Please confirm receipt of this request and ensure all relevant staff members are notified of these privacy preferences.

Sincerely,
${parentName}
Parent/Legal Guardian of ${studentName}`;
}
